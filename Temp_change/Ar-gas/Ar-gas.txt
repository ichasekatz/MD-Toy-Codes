units metal
atom_style atomic
boundary p p p

region box block 0 100 0 100 0 100
create_box 1 box

create_atoms 1 random 100 12345 box
mass 1 39.948

pair_style lj/cut 2.5
pair_coeff * * 0.0103 3.4

group Argas region box

timestep 0.01
velocity Argas create 2000 12345 mom yes rot yes

thermo 25
thermo_style custom step temp press etotal vol

variable step equal step
variable temp equal temp
variable press equal press
variable etotal equal etotal
variable vol equal vol

fix samp_out all print 10 "${step} ${temp} ${press} ${etotal} ${vol}" &
                 file thermo_output.txt screen no

fix 1 all nvt temp 2000 100 0.5
dump heat all atom 50 Ar-gas.dump
run 10000
